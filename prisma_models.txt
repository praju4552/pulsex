--- model PrototypingOrder ---
  id              String                  @id @default(uuid())
  orderRef        String                  @unique @default(cuid())
  salesOrderId    String?                 @unique
  invoiceId       String?                 @unique
  projectId       String?                 @unique
  jobId           String?                 @unique
  firstName       String
  lastName        String
  email           String
  phone           String
  streetAddress   String
  apartment       String?
  city            String
  state           String
  zip             String
  country         String
  serviceType     String   // "PCB Printing" | "3D Printing"
  specifications  Json     // Full PCBSpec object or 3D spec
  specSummary     String   // Human-readable summary for admin
  shippingMethod  String
  shippingCost    Int
  pcbPrice        Int
  totalAmount     Int
  orderStatus     PrototypingOrderStatus @default(PENDING)
  paymentStatus   PaymentStatus          @default(UNPAID)
  adminNotes      String?
  userId          String?
  user            PrototypingUser? @relation(fields: [userId], references: [id])
  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt

  credits     Int      @default(0) // Moved from old User model
  failedLoginAttempts Int       @default(0)
  lockedUntil         DateTime?
  streetAddress String?
  apartment     String?
  city          String?
  state         String?
  zip           String?
  country       String?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
  orders        PrototypingOrder[]
  threeDFiles   ThreeDFile[]
  threeDOrders  ThreeDOrder[]
  laserFiles    LaserCuttingFile[]
  laserOrders   LaserCuttingOrder[]

--- model ThreeDFile ---
  id           String   @id @default(uuid())
  userId       String?
  filePath     String
  fileType     String
  fileSize     Int
  uploadStatus String   @default("PENDING") // PENDING, PROCESSING, COMPLETED, FAILED
  createdAt    DateTime @default(now())
  user         PrototypingUser? @relation(fields: [userId], references: [id])
  metadata     ThreeDMetadata?
  config       ThreeDPrintConfig?
  order        ThreeDOrder?

--- model ThreeDMetadata ---
  id                 String   @id @default(uuid())
  fileId             String   @unique
  volume             Float?
  surfaceArea        Float?
  width              Float?
  height             Float?
  depth              Float?
  triangleCount      Int?
  estimatedPrintTime Float?   // In minutes
  screenshotUrl      String?
  file               ThreeDFile @relation(fields: [fileId], references: [id], onDelete: Cascade)

--- model ThreeDPrintConfig ---
  id                 String   @id @default(uuid())
  fileId             String   @unique
  material           String
  infill             Int
  scale              Float    @default(1.0)
  layerHeight        Float
  finish             String
  color              String?
  file               ThreeDFile @relation(fields: [fileId], references: [id], onDelete: Cascade)

--- model ThreeDOrder ---
  id                 String                 @id @default(uuid())

  configId           String?

  quantity           Int                    @default(1)
  status             PrototypingOrderStatus @default(PENDING)

  updatedAt          DateTime               @updatedAt
  user               PrototypingUser?       @relation(fields: [userId], references: [id])
  file               ThreeDFile             @relation(fields: [fileId], references: [id])

--- model LaserCuttingFile ---
  id           String   @id @default(uuid())
  userId       String?
  filePath     String
  fileName     String
  fileType     String   // dxf, svg, pdf, ai, png, jpg, bmp
  fileSize     Int
  uploadStatus String   @default("UPLOADED") // UPLOADED, PROCESSING, COMPLETED, FAILED
  createdAt    DateTime @default(now())
  user         PrototypingUser? @relation(fields: [userId], references: [id])
  config       LaserCuttingConfig?
  order        LaserCuttingOrder?

--- model LaserCuttingConfig ---
  id           String @id @default(uuid())
  fileId       String @unique
  serviceType  String // "cutting" | "engraving" | "both"
  material     String // wood_mdf, wood_plywood, hardwood, acrylic_cast, acrylic_extruded, leather, fabric, cork, felt, rubber
  thickness    Float  // in mm
  width        Float  // in mm (design area)
  height       Float  // in mm (design area)
  quantity     Int    @default(1)
  finish       String @default("standard") // standard, polished, painted
  urgency      String @default("standard") // standard, rush_24h
  file         LaserCuttingFile @relation(fields: [fileId], references: [id], onDelete: Cascade)

--- model LaserCuttingOrder ---
  id        String                 @id @default(uuid())

  configId  String?

  status    PrototypingOrderStatus @default(PENDING)
  createdAt DateTime               @default(now())
  updatedAt DateTime               @updatedAt
  user      PrototypingUser?       @relation(fields: [userId], references: [id])
  file      LaserCuttingFile       @relation(fields: [fileId], references: [id])
